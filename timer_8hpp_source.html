18/**
19 * @brief Class that represents timer.
20 * @ingroup Chrono timers
21 * @warning This class behaviour is different with and without `GOLXZN_MULTITHREADING` macros definition.
22 * @tparam OnTimerDone type of function that will be called when timer is done.
23 * @tparam BaseClock clock that will be used for measurement. It has to be monotonic and STL compatible.
24 *
25 * This class is used to measure time. It calls callback when it's done.
26 * If `GOLXZN_MULTITHREADING` __is not__ defined it doesn't use a thread.
27 * So to call callback it has to be updated in main thread. It storing callback.
28 * If `GOLXZN_MULTITHREADING` __is__ defined it uses a thread.
29 * So to call callback it doesn't have to be updated. It doesn't storing callback, but std::thread
30 *
31 * Example of using:
32 * @code{.cpp}
33 * std::atomic_bool executed{ false };
34 * golxzn::os::chrono::timer timer2{ 10ms,
35 * std::function<void()>{ [&executed] { executed.store(true); } }
36 * };
37 *
38 * while(!timer2.is_done()) {
39 * std::this_thread::sleep_for(1ms);
40 * }
41 * @endcode
42 */
43template<class OnTimerDone, class BaseClock = utils::default_base_clock>
44class timer {
45 static_assert(BaseClock::is_steady,
46 "[golxzn::os::chrono::timer] BaseClock is not a monotonic clock");
47 static_assert(utils::enough_resolution_v<BaseClock>,
48 "[golxzn::os::chrono::timer] BaseClock's resolution is less than microseconds!");
49
50public:
51 using base_clock = BaseClock; ///< Base clock type
52 using time_point = typename base_clock::time_point; ///< Type of time point from base_clock
53 using timer_end_callback = OnTimerDone; ///< Type of callback that will be called when timer is done

55 /**
56 * @brief Default precision of timer.
57 * > Only if `GOLXZN_MULTITHREADING` is defined.
58 * This value is used as time to wait inside timer thread.

60 * while(is_running()) {
61 * std::this_thread::sleep_for(default_precision);
62 * }
63 * @endcode
64 */

142/**
143 * @brief Class that represents fast_timer.
144 * @details Compared to golxzn::os::chrono::timer it doesn't use a thread and callback.
145 * @ingroup Chrono timers
146 * @tparam BaseClock clock that will be used for measurement. It has to be monotonic and STL compatible.
147 *
148 * This class is used to measure time. It doesn't use a thread and callback.
149 * It stores only start time and timer interval, which is passing in constructor.
150 * So it's very lightweight. It's just check if the creating time + interval is less than current time.
151 *
152 * Example of using:
153 * @code{.cpp}
154 * golxzn::os::chrono::fast_timer<> timer{ 100ms }
155 * while (timer.is_running()) {
156 * const auto left{ timer.time_left() };
157 * // do something
158 * }
159 * @endcode
160 */

164 "[golxzn::os::chrono::fast_timer] BaseClock is not a monotonic clock");
